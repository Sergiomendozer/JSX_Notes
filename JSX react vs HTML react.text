##JSX React:

import React from 'react';

const Hello = () => {
  return (
    <div>
      <h1>Hello Vishwas</h1>
    </div>
  );
}

export default Hello;

##HTML React:
import React from 'react';

const Hello = () => {
  // Using React.createElement
  return React.createElement(
    'div',
    { id: 'hello', className: 'dummyClass' },
    React.createElement('h1', null, 'Hello Vishwas')
  );
}

export default Hello;
